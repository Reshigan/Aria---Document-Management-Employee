
    def _post_remittance_to_sap(self, document_data: Dict[str, Any]) -> Dict[str, Any]:
        """Post remittance advice to SAP using F-53 transaction (Incoming Payments)"""
        try:
            business_data = document_data.get('business_data', {})
            amounts = business_data.get('amounts_found', [])
            dates = business_data.get('dates_found', [])
            
            # Extract remittance data from OCR text
            ocr_text = document_data.get('ocr_text', '')
            
            # Look for payment reference
            payment_ref = None
            for line in ocr_text.split('\n'):
                if 'payment reference' in line.lower() or 'reference' in line.lower():
                    parts = line.split(':')
                    if len(parts) > 1:
                        payment_ref = parts[1].strip()
                        break
            
            # Look for bank details
            bank_account = None
            for line in ocr_text.split('\n'):
                if 'account' in line.lower() and any(char.isdigit() for char in line):
                    parts = line.split(':')
                    if len(parts) > 1:
                        bank_account = parts[1].strip()
                        break
            
            # Prepare SAP remittance document
            sap_document = {
                'transaction_code': 'F-53',      # Incoming Payments
                'document_type': 'DZ',           # Payment Document
                'company_code': self.config.get('company_code', '1000'),
                'posting_date': dates[0] if dates else time.strftime('%Y-%m-%d'),
                'document_date': dates[0] if dates else time.strftime('%Y-%m-%d'),
                'reference': payment_ref or f"REM-{int(time.time())}",
                'document_header_text': f"Remittance Advice - {document_data.get('filename', 'Unknown')}",
                'line_items': []
            }
            
            # Add payment line items
            total_amount = 0
            if amounts:
                try:
                    # Clean and parse amounts
                    for amount_str in amounts:
                        clean_amount = ''.join(c for c in amount_str if c.isdigit() or c in '.,')
                        if clean_amount:
                            amount = float(clean_amount.replace(',', ''))
                            if amount > total_amount:
                                total_amount = amount
                except:
                    total_amount = 0
            
            # Bank clearing line item
            sap_document['line_items'].append({
                'line_type': 'bank_clearing',
                'account': bank_account or '113100',  # Bank clearing account
                'amount': total_amount,
                'debit_credit': 'D',  # Debit for incoming payment
                'text': f"Payment received - {payment_ref or 'Remittance'}"
            })
            
            # Customer clearing line item (credit)
            sap_document['line_items'].append({
                'line_type': 'customer_clearing',
                'account': '140000',  # Customer clearing account
                'amount': total_amount,
                'debit_credit': 'C',  # Credit to clear customer balance
                'text': f"Customer payment - {payment_ref or 'Remittance'}"
            })
            
            # Mock SAP API call for remittance
            sap_response = {
                'success': True,
                'sap_document_number': f"5100{int(time.time()) % 100000}",
                'transaction_code': 'F-53',
                'posting_date': sap_document['posting_date'],
                'total_amount': total_amount,
                'currency': 'ZAR',
                'message': f"Remittance advice posted successfully to SAP"
            }
            
            logger.info(f"Remittance posted to SAP: {sap_response['sap_document_number']}")
            return sap_response
            
        except Exception as e:
            logger.error(f"Error posting remittance to SAP: {str(e)}")
            return {
                'success': False,
                'error': str(e),
                'sap_document_number': None,
                'transaction_code': 'F-53'
            }
